\section{Power-based number system for toki pona}
	
\textit{Written by jan Loje with the help of jan Tamalu and Shaevor}
\textit{(mistakes are mine only)}

This system should be:

1. based on power-of-10 notation, a.k.a. scientific notation, 

2. easy to understand, learn, and use,

3. suitable for \textit{toki pona}.

NOTES: < >: \textit{read as}

\section{Rationale}

\vspace{5pt}

This system is a way to \textit{read} numbers and dates written with the digits (0-9) in \textit{toki pona} text. Additional meanings are added to some already existing \textit{toki pona} words.


\section{Use}
	
\subsection{Prefixes (when needed)}

    \textbf{Ordinal and cardinal numbers}

	\vspace{3 pt}
	
	{\sitp{nanpa}} <\textsf{nanpa}>  ordinal number
	
    \vspace {3 pt}
	{\sitp{mute}} <\textsf{mute}> cardinal number
	
	\vspace{5 pt}
	
	{\sitp{ona li nanpa 5}}~<\textsf{ona li nanpa luka}> \textit{it's the 5th} (ordinal)

    \vspace {3 pt}	
	{\sitp{ona li mute 5}}~<\textsf{ona li mute luka}> \textit{it's 5} (cardinal) 
	
	
\subsection{Positional digits}
	
	\textbf{The values of digits are \textit{positional} (common usage)}

	\vspace{5 pt}
		
    That is 212 = 2 × 10\textsuperscript{2} + 1 × 10\textsuperscript{1} + 2 × 10\textsuperscript{0}
	
	\vspace{5 pt}
	
	12 <\textsf{wan tu}>
	
	\vspace{5 pt}
	2024 <\textsf{tu ala tu lipu}>
	
\subsection{Numbers as powers of 10}
    
   \textbf{ \textsf{sewi} is the base 10 for all powers.}
    
    \vspace{5 pt}
	
	1000 = 10\textsuperscript{3} <\textsf{sewi sin}> \textit{one thousand}
	
   \vspace {6 pt}
	10,000 = 10\textsuperscript{4} <\textsf{sewi lipu}> \textit{ten thousand}
	
	...
	
	10,000,000 = 10\textsuperscript{7} \textsf{<sewi len>} \textit{a crore}
	
	....
	
	1 000 000 000 = 10\textsuperscript{9} <\textsf{sewi suli}> \textit{a billion}
	
\vspace {12 pt}
	
	{\sitp{jan li jo e \$1,000,000,000}}~<\textsf{jan li jo e mani Mewika pi mute sewi suli}>
	
\subsection{Very large (or small) numbers}

   \textbf{ Very large (or small) numbers can be expressed easily.}
    
    \vspace {6 pt}
	
	 10\textsuperscript{100} \textsf{<sewi wan ala ala>} \textit{one googol}
	 
	 10\textsuperscript{10\textsuperscript{2}} <\textsf{sewi sewi tu}> \textit{one googol}
	
	\subsection{Composed numbers}
	
	\textbf{Numbers with multiplicative and additive values.}
	
	\vspace {6 pt}
	
	The number to the left of \textit{sewi} has multiplicative value.
	
		\vspace{5pt}
	
	The additive value of a number (sequence) is stated explicitly with
	\textit{en}.
     \vspace {6 pt}
   	
	4,000,000,012 = 4 × 10\textsuperscript{9}+ 12  <\textsf{lipu sewi suli en	wan tu}>
	
	\subsection{Numbers with fractional parts}
	
	\textbf{Number with a fractional part separated by a decimal point.}
	
		\vspace{5pt}
		
	3.14 <\textsf{sin sike wan lipu}>
	
		\vspace{5pt}
		
	3.14 = 314 × 10\textsuperscript{-2} <\textsf{sin wan lipu sewi weka tu}>
	
	\subsection{Numbers with negative exponents}
	
	\textbf{Negative exponents are prefixed by \textit{weka}.}

		\vspace{5pt}
	
	6.62 × 10\textsuperscript{-34} <\textsf{pipi sike pipi tu sewi weka sin lipu}>
	
	\subsection{Dates}
	
	\textbf{ISO 8601 system}
	\vspace{5pt}
	
	{\sitp{2024-05-12}}~<\textsf{tenpo sike tu ala tu lipu \textbf{en} tenpo mun luka \textbf{en} tenpo suno wan tu}>
	
	\vspace{5pt}
	
	{\sitp{5-12 la ona li kama lon}}~<\textsf{tenpo mun luka en tenpo suno wan tu la ona li kama lon ale}> His birthday is May 12th
